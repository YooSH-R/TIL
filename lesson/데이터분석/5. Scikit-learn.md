# 정확도
```py
from sklearn.metrics import accuracy_score
from sklearn.tree import DecisionTreeClassifier
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split

X, y = load_iris(return_X_y=True)
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

model = DecisionTreeClassifier(random_state=42)
model.fit(X_train, y_train)

# 방법 1: accuracy_score (예측값을 직접 비교)
y_pred = model.predict(X_test)
acc1 = accuracy_score(y_test, y_pred)

# 방법 2: model.score (내부에서 예측 + 비교를 한번에)
acc2 = model.score(X_test, y_test)

print(f"accuracy_score: {acc1}")
print(f"model.score:    {acc2}")
print(f"결과 일치: {acc1 == acc2}")
```

# 오차 행렬
- 위양성, 위음성, 진양성, 진음성
- 재현율: 진양성 / 양성, 정밀도: 진양성 / (양성이라고 진단)
- 암검사 처럼 놓치면 안 되는 경우: 재현율(Recall) 중시, 스팸 필터처럼 오진 싫은 경우: 정밀도(Precision) 중시
- F1-score: 정밀도와 재현율의 조화 평균

```py
from sklearn.metrics import confusion_matrix
import numpy as np

y_true = [1, 0, 1, 1, 0, 1]
y_pred = [0, 0, 1, 1, 0, 1]

# 오차 행렬 출력
cm = confusion_matrix(y_true, y_pred)
print(cm)

# [[2, 0],  <- 0(정상)을 맞춘/틀린 개수
#  [1, 3]]  <- 1(암)을 맞춘/틀린 개수 (여기서 1을 놓친 게 FN)

# 클래스별로 정밀도, 재현율, F1-score 등 보여줌
print(classification_report(y_true, y_pred))
```
