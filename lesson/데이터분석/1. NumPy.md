# import
- import numpy as np

# 벡터화 연산
- np.array([1, 2, 3]) + np.array([1, 2, 3])

# 배열의 구조
- arr.ndim: 데이터 차원(1, 2, 3)
- arr.shape: 데이터 모양(가로, 세로 길이)
- arr.dtype: 타임(배열 안 알맹이가 정수인지 실수인지 등)

# 배열 생성
- np.array([1.3, 2.7], dtype='int')
- np.zeros()
- np.ones()
- np.full((2, 2), 7): 2x2 행렬 7로 채우기
- np.arange(10)
- np.linspace(0, 10, 5): 0부터 10(포함)까지, 5개 점으로 나누기
- np.random.seed(42): 시드 심기
- np.random.rand(3): 0과 1 사이의 균일 분포에서 뽑기
- np.random.randn(3): 평균 0 표준편차 1인 정규 분포에서 뽑기
- np.random.randint(1, 101, size=5): 범위 내 정수 size만큼 뽑기
- np.random.choice(리스트): 리스트에서 무작위로 뽑기

# 데이터 타입
- arr.astype('int')

# 배열 접근
- arr[1, 2]: 1행 2열
- arr[1:, 1:]: 1행부터 끝까지, 1열부터 끝까지
- arr[arr >= 60]
- arr[[0, 2, 0, 3]]: 리스트 값만큼 인덱스에서 찾아옴
- arr[0] = 222: 데이터 수정
- arr[arr == 0] = 1: 데이터 수정

# 배열 형태 변환
- arr.reshape(3, 4): 3x4 행렬로 변환
- arr.reshape(-1, 3): nx3 행렬로 자동 계산
- arr.flatten(): 데이터 복사해서 평탄화
- arr.ravel(): 데이터 참조해서 평탄화(arr.reshqpe(-1) 과 같음)
- arr.T: 전치 행렬
- np.vstack((arr1, arr2))
- np.hstack((arr1, arr2))

# 사칙 연산과 행렬 곱
- arr1 + arr2
- arr1 * arr2: 그냥 곱셈
- arr1 @ arr2: 행렬 곱셈

# 통계 연산
- arr.sum()
- arr.mean()
- arr.연산(axis=0 또는 1): 열/행별 통계

# 위치 찾기
- np.max(arr): 값
- np.argmax(arr): 인덱스

# 브로드캐스팅(알아서 모양 맞춰주기)
- 규칙: 뒤에서부터 숫자 맞춰본다
- (3, 3) + (3, ): 가능
- (3, 3) + (1, 3): 가능
- (3, 3) + (2, 3): 불가능

# 조건부 연산
- np.where(조건, 참일 때 값, 거짓일 때 값)

# 집합과 유니크
- types, counts = np.uique(arr, return_counts=True)
- 중복 제거하고 오름차순 정렬된 결과

# 선형대수
- np.linalg.norm(arr1 - arr2): 유클리드 거리
- arr1 @ arr2: 내적 -> 유사도
- np.linalg.solve(arr1, arr2): 연립1차 방정식 풀이
- np.linalg.inv(arr): 역행렬
