# import
- import seaborn as sns
- import matplotlib.pyplot as plt

# 히스토그램
- 데이터가 어디에 많이 몰려있는지 알고 싶을 때, ex) 점수 분포, 키 분포, 나이 분포
- df = sns.load_dataset('titanic'): Seaborn 내장 예제 불러오기
- sns.histplot(daga=df, x='age', kde=True): 데이터프레임, 컬럼명, 부드러운 곡선(밀도 추정) 그리기
- plt.title("히스토그램 제목")
- plt.show()

# 박스 플롯
- 데이터의 분포와 이상치(Outlier)를 한눈에 보여줌
- df = sns.load_dataset('titanic')
- sns.boxplot(data=df, x='class', y='age')
- plt.title("박스 플롯 제목")
- plt.show()

# 카운트 플롯
- 범주형 데이터가 각각 몇 개인지 막대 그래프로 보여줌
- df = sns.load_dataset('titanic')
- sns.countplot(data=df, x='class')
- plt.title("제목")
- plt.show()

# 막대 그래프 심화
- 데이터를 주면, 자동으로 평균을 계산해 막대 높이 결정하고, 막대 위에 검은색 선으로 신뢰 구간 표시
- df = sns.load_dataset('titanic')
- sns.barplot(data=df, x='class', y='survived', hue='sex'): hue-색깔로 그룹을 한 번 더 나눔
- plt.title("제목")
- plt.show()

# 히트맵
- 데이터의 값을 색상으로 표현하는 그래프, 주로 여러 변수들 간 상관관계를 한눈에 파악할 때 사용
- df = sns.load_dataset('titanic')
- numerical_df = df.select_dtypes(include=['number']): 수치형 데이터만 선택
- corr_matrix = numerical_df.corr(): 상관관계 계산
- sns.heatmap(data=corr_matrix, annot=True, cmap='coolwarm'): annot-숫자를 같이 표시, cmap-색상 테마를 설정
- plt.title("제목")
- plt.show()

# 페어 플롯
- 데이터에 있는 모든 변수를 짝지어서 한번에 보여줌
- df = sns.load_dataset('iris')
- sns.pairplot(df, hue='species')
- plt.show()

# 서브 플롯
- 화면을 여러 개로 쪼개서 한 번에 여러 그래프를 비교
- df = sns.load_dataset('titanic')
- fig, axes = plt.subplots(1, 2, figsize=(10, 5)): 1줄, 2칸 짜리 도화지 생성
- sns.histplot(data=df, x='age', ax=axes[0]): axes는 인덱스로 접근
- axes[0].set_title("제목")
- sns.boxplot(data=df, x='class', y='age', ax=axes[1])
- axes[1].set_title("제목")
- plt.tight_layout(): 그래프 요소들이 겹치지 않도록 자동으로 간격을 조정
- plt.show()
